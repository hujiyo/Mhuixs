对象管理：【HOOK】
对象类型：【TABLE,KVALOT,LIST,BITMAP,STREAM】
操作关键字：【DEL,TYPE,RANK,CLEAR,DESC,GET,WHERE,TEMP,SET,EXISTS,SELECT,KEY,INSERT,FIELD,ADD,SWAP,ATTRIBUTE,POS,APPEND,LEN,LPUSH,RPUSH,LPOP,RPOP,COUNT,COPY,MOVE,MERGE,SPLIT,EXPORT,IMPORT,SUBSCRIBE,UNSUBSCRIBE,NOTIFY,LOCK,UNLOCK,COMMIT,ROLLBACK,ASYNC,SYNC,WAIT】
类型关键字【i1/int8_t,i2/int16_t,i4/int32_t/int,i8/int64_t,ui1/uint8_t,ui2/uint16_t,ui4/uint32_t,ui8/uint64_t,f4/float,f8/double,str/string,date,time,datetime,bool,blob,json】
键关键字【PKEY,FKEY,UNIQUE,NOTNULL,DEFAULT,INDEX,AUTO_INCREMENT】
宏替换：【$】
条件关键字：【IF,ELSE,ENDIF,WHILE,ENDWHILE,FOR,ENDFOR,BREAK,CONTINUE】
逻辑关键字：【AND,OR,NOT,IN,BETWEEN,LIKE,IS,NULL】

基础语法
[GET objname;] 	#获取HOOK对象的所有数据
[WHERE;] 	#返回当前操作对象的基本信息
[DESC;] 	#返回当前操作对象的视图信息
[HOOK;] 	#回归主HOOK根，此HOOK为STREAM类型临时HOOK,名字为用户名,存储用户描述信息
[HOOK objtype objname1 objname2 ...;] #使用钩子创建操作对象
[HOOK objname;] #手动切换到一个已经存在的操作对象
[HOOK DEL objname1 objname2 ...;] #注销指定的HOOK操作对象
[HOOK CLEAR objname1 objname2 ...;] #清空当前操作对象的所有数据,但保留操作对象及其权限信息
[HOOK COPY src_objname dst_objname;] #复制HOOK对象及其数据
[HOOK SWAP src_objname dst_objname;] #交换HOOK对象
[HOOK MERGE src_objname dst_objname;] #合并多个HOOK对象(只支持相同类型之间的合并)
[RANK objname rank;] #设置HOOK操作对象的权限信息
[GET RANK objname;] #获取指定HOOK操作对象的权限等级
[GET TYPE objname;] #获取指定HOOK操作对象的类型
[LOCK objname timeout;] #锁定HOOK对象，timeout为超时时间(秒)
[UNLOCK objname;] #解锁HOOK对象
[EXPORT objname format;] #导出HOOK对象数据，format: json/csv/binary
[IMPORT objname format data;] #导入数据到HOOK对象
[HOOK RENAME old_key new_key;] #重命名键
[CHMOD objname mode;] #设置HOOK对象的权限信息
[GET CHMOD objname;] #获取指定HOOK操作对象的权限等级
[WAIT timeout;] #等待指定时间(毫秒)（上一个命令执行完毕并传回客户端后客户端开始执行）(事务内使用默认忽略此命令)

# 事务控制
[MULTI;]#开始事务
[EXEC;]#提交事务
[ASYNC command;] #异步执行命令
[SYNC;] #同步等待所有异步操作完成


# 条件控制
[IF condition;] #条件判断开始
[ELSE;] #否则分支
[ELIF condition;] #else if
[WHILE condition;] #循环开始
[FOR var start end step;] #for循环
[END;] #循环结束/条件判断结束/for循环结束
[BREAK;] #跳出循环
[CONTINUE;] #继续下一轮循环

TABLE类语法 不指明FIELD默认就是LINE操作
[FIELD ADD field_name datatype restraint ;] #增加一个字段
[FIELD INSERT index field_name datatype restraint;] #只支持单次插入一个字段
[FIELD SWAP field1_index field2_index;] #交换字段
[FIELD DEL field1_index field2_index ...;] #删除指定字段
[FIELD SET ATTRIBUTE field_index attribute;] #设置字段约束性属性
[FIELD GET INFO field_index;] #获取字段详细信息
[ADD value1 value2...;] #增加一条记录
[INSERT line_index value1 value2 ...;] #插入一条记录
[SET line1_index field1_index value1 line2_index field2_index value2 ...;] #修改某行的部分信息
[DEL line1_index line2_index;] #删除line1_index到line2_index的行，可以相同
[SWAP line1_index line2_index;] #交换两条记录
[GET line1_index line2_index;] #获取行号之间的多行数据，可以相同
[GET FIELD field_name;] #获取指定字段对应的列的数据
[GET POS x1 y1 x2 y2 ...;]#标准 POS_GET 语句,获取指定坐标的数据
[GET COUNT;] #获取记录总数
[GET WHERE condition;] #根据条件查询记录
[SET WHERE condition field_name value;] #根据条件批量更新字段值
[DEL WHERE condition;] #根据条件批量删除记录
[SORT field_name order;] #按字段排序，order: ASC/DESC
[INDEX CREATE field_name index_type;] #创建指定字段索引
[INDEX DEL field_name;] #删除指定字段索引


KVALOT类语法
[EXISTS key1 key2 ...;] #判断存在几个键
[SELECT pattern;] #查找所有符合给定模式的键,注意查询的是键，不是值,pattern是字符串
[SET TYPE type key1 key2 key3 ... ;] #创建新键
[SET key value;] #设置键值对（仅支持STREAM/BITMAP）
[GET key;] #获取键对应的值（仅支持STREAM/BITMAP）
[DEL key1 key2 ...;] #删除指定键
[KEY key;]# 进入键对象操作
[COPY src_key dst_key;] #复制键值对
[SWAP src_key dst_key;] #移动键值对
[RENAME old_key new_key;] #重命名键
[EXPIRE key microseconds;] #设置键的过期时间
[PERSIST key;] #移除键的过期时间
[TTL key;] #获取键的剩余生存时间
[INCR key;] #将键的值递增1
[DECR key;] #将键的值递减1
[INCR key value;] #将键的值递增指定数值
[DECR key value;] #将键的值递减指定数值
[GET KEYS;] #获取所有键名
[GET COUNT;] #获取键的总数

STREAM类语法
[APPEND value;]#将数据追加向流中附加数据
[APPEND pos value;]#将value追加到流中，从指定位置开始。若pos超出流的长度，则从流的末尾开始追加。
[GET pos len;]#获取流中指定长度的数据
[SET pos value;]#从pos处设置流中指定位置的值
[SET pos len char;]#从pos处设置流中指定位置的值,并将其长度设置为len
[GET LEN;] #获取流的长度
[GET;] #获取流的所有数据
[CLEAR;] #清空流
[MERGE src_stream;] #合并另一个流

LIST类语法
[LPUSH value;]#标准 LIST_LPUSH 语句，在列表开头添加一个值
[RPUSH value;]#标准 LIST_RPUSH 语句，在列表末尾添加一个值
[LPOP;]#标准 LIST_LPOP 语句，移除并返回列表开头的值
[RPOP;]#标准 LIST_RPOP 语句，移除并返回列表末尾的值
[GET index;] #标准 LIST_GET 语句,获取指定位置的值,index:1,2..为第1,2..个元素，-1,-2,为倒数第1,2..个元素，0:所有元素
[DEL index;] #标准 LIST_DEL 语句，删除指定索引位置的值
[GET LEN index;]#标准 LIST_GET_INDEX_LEN 语句，获取列表中指定索引位置的值的长度
[INSERT index value;] #标准 LIST_INSERT 语句，在指定索引位置插入一个值
[SET index value;]#标准 LIST_SET 语句，更新列表中指定索引位置的值
[EXISTS value1 value2 ...;]#标准 LIST_EXISTS 语句，判断列表是否存在指定值
[GET LEN;] #标准 LIST_GET_LEN 语句，获取列表的长度
[SWAP index1 index2;] #交换两个位置的值
[COPY src_index dst_index;] #复制元素到指定位置
[CLEAR;] #清空列表
[REVERSE;] #反转列表
[SORT order;] #排序列表，order: ASC/DESC
[UNIQUE;] #去重
[GET start end;] #返回切片操作
[FIND value;] #查找值的索引位置
[COUNT value;] #统计值的出现次数
[JOIN src_list;] #合并另一个列表

BITMAP类语法
[SET offset value;] #设置位图中指定偏移量处的位值，value为0或1
[SET offset1 offset2 value;] #设置位图中指定偏移量范围内的位值，value为0或1
[GET offset;] #获取位图中指定偏移量处的位值
[GET offset1 offset2;]#获取位图中指定偏移量范围内的位值
[COUNT offset1 offset2;] #统计位图中指定偏移量范围内值为1的位数
[COUNT;] #统计整个位图中值为1的位数
[FLIP offset;] #翻转指定偏移量处的位值
[FLIP offset1 offset2;] #翻转指定偏移量范围内的位值
[CLEAR;] #清空位图(全部置0)
[FILL value;] #填充位图(全部置为指定值)
[FIND value;] #查找第一个指定值的位置
[FIND value start;] #从指定位置开始查找第一个指定值的位置
[AND src_bitmap;] #与另一个位图进行AND操作
[OR src_bitmap;] #与另一个位图进行OR操作
[XOR src_bitmap;] #与另一个位图进行XOR操作
[NOT;] #对位图进行NOT操作
[SHIFT direction count (MATHS);] #位移操作，direction: LEFT/RIGHT,默认是算术位移，如果需要逻辑位移，则需要加上MATHS关键字
[GET SIZE;] #获取位图大小
[RESIZE size;] #调整位图大小（用0填充）
[IMPORT format data;] #导入位图数据
[JOIN src_bitmap;] #合并另一个位图

# 跨HOOK互动操作
[HOOK JOIN hook1 hook2 new_hook;] #合并两个HOOK对象的数据
[KEY JOIN key1 key2 new_key;] #合并两个键的数据

# 系统管理
[SYSTEM INFO;] #获取系统信息
[SYSTEM STATUS;] #获取系统统计信息
[SYSTEM REGISTER;] #获取所有HOOK对象列表(注册表仅root用户可见)
[SYSTEM CLEANUP;] #清理系统资源
[SYSTEM BACKUP path;] #系统备份
[SYSTEM RESTORE path;] #系统恢复
[SYSTEM LOG level;] #打印级别系统日志（INFO/WARN/ERROR）

#新增
[SYSTEM SHUTDOWN;] #系统关闭

# 宏和变量（客户端解析器本地操作）
[$var_name value;] #定义变量
[GET $var_name;] #获取变量值
[SET $var_name value;] #设置变量值
[DEL $var_name;] #删除变量

#持久性和压缩管理
[BACKUP objname;] #备份对象
[ZS objname rank;] #主动压缩数据
